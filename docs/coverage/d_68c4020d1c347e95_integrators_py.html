30def add_inner_nonbonded_force(system, inner_switch, inner_cutoff, force_group_index): 

31 """To a given OpenMM System_ containing a NonbondedForce_ object, this function adds 

32 a new force group with the purpose of performing multiple time-scale integration 

33 according to the RESPA2 splitting scheme of Morrone, Zhou, and Berne 

34 :cite:`Morrone_2010`. Besides, it assigns the provided `force_group_index` to this 

35 new group and `force_group_index+1` to the original NonbondedForce_. When used in 

36 any instance of :class:`AbstractMiddleRespaIntegrator`, the new force group must be 

37 identified as being embodied by the NonbondedForce_ as opposed to being 

38 complimentary to it. 

39 

40 .. warning: 

41 The new force group is not intended to contribute to the system energy. Its sole 

42 purpose is to provide a smooth, short-range force calculator for some 

43 intermediary time scale in a RESPA-type integration. 

44 

45 Parameters 

46 ---------- 

47 system : openmm.System 

48 The system the inner force will be added to, which must contain a 

49 NonbondedForce_. 

50 inner_switch : float or unit.Quantity 

51 The inner switching distance, where the interaction of an atom pair begins 

52 to switch off to zero. 

53 inner_cutoff : float or unit.Quantity 

54 The inner cutoff distance, where the interaction of an atom pairs completely 

55 switches off. 

56 force_group_index : int 

57 The force group the new interactions will belong to. The old NonbondedForce_ 

58 will be automatically assigned to `force_group_index+1`. 

59 

60 Example 

61 ------- 

62 >>> import ufedmm 

63 >>> from openmm import unit 

64 >>> dt = 2*unit.femtoseconds 

65 >>> temp = 300*unit.kelvin 

66 >>> tau = 10*unit.femtoseconds 

67 >>> gamma = 10/unit.picoseconds 

68 >>> model = ufedmm.AlanineDipeptideModel() 

69 >>> ufedmm.add_inner_nonbonded_force( 

70 ... model.system, 5*unit.angstroms, 8*unit.angstroms, 1 

71 ... ) 

72 """ 

73 if openmm.__version__ < "7.5": 

74 raise Exception("add_inner_nonbonded_force requires OpenMM version >= 7.5") 

75 try: 

76 nonbonded_force = next( 

77 filter(lambda f: isinstance(f, openmm.NonbondedForce), system.getForces()) 

78 ) 

79 except StopIteration: 

80 raise Exception("add_inner_nonbonded_force requires system with NonbondedForce") 

81 if ( 

82 nonbonded_force.getNumParticleParameterOffsets() > 0 

83 or nonbonded_force.getNumExceptionParameterOffsets() > 0 

84 ): 

85 raise Exception("add_inner_nonbonded_force does not support parameter offsets") 

86 periodic = nonbonded_force.usesPeriodicBoundaryConditions() 

87 rs = _standardized(inner_switch) 

88 rc = _standardized(inner_cutoff) 

89 a = rc + rs 

90 b = rc * rs 

91 c = (30 / (rc - rs) ** 5) * np.array( 

92 [b**2, -2 * a * b, a**2 + 2 * b, -2 * a, 1] 

93 ) 

94 f0s = sum([c[n] * rs ** (n + 1) / (n + 1) for n in range(5)]) 

95 

96 def coeff(n, m): 

97 return c[m - 1] if m == n else c[m - 1] / (m - n) 

98 

99 def func(n, m): 

100 return "*log(r)" if m == n else (f"*r^{m-n}" if m > n else f"/r^{n-m}") 

101 

102 def val(n, m): 

103 return f0s if m == 0 else (coeff(n, m) - coeff(0, m) if n != m else coeff(n, m)) 

104 

105 def sgn(n, m): 

106 return "+" if m > 0 and val(n, m) >= 0 else "" 

107 

108 def S(n): 

109 return "".join(f"{sgn(n, m)}{val(n, m)}{func(n, m)}" for m in range(6)) 

110 

111 potential = "eps4*((sigma/r)^12-(sigma/r)^6)+Qprod/r" 

112 potential += ( 

113 f"+step(r-{rs})*(eps4*(sigma^12*({S(12)})-sigma^6*({S(6)}))+Qprod*({S(1)}))" 

114 ) 

115 mixing_rules = "; Qprod=Q1*Q2" 

116 mixing_rules += "; sigma=halfsig1+halfsig2" 

117 mixing_rules += "; eps4=sqrt4eps1*sqrt4eps2" 

118 force = openmm.CustomNonbondedForce(potential + mixing_rules) 

119 for parameter in ["Q", "halfsig", "sqrt4eps"]: 

120 force.addPerParticleParameter(parameter) 

121 force.setNonbondedMethod( 

122 force.CutoffPeriodic if periodic else force.CutoffNonPeriodic 

123 ) 

124 force.setCutoffDistance(inner_cutoff) 

125 force.setUseLongRangeCorrection(False) 

126 ONE_4PI_EPS0 = 138.93545764438198 

127 for index in range(nonbonded_force.getNumParticles()): 

128 charge, sigma, epsilon = map( 

129 _standardized, nonbonded_force.getParticleParameters(index) 

130 ) 

131 force.addParticle( 

132 [charge * np.sqrt(ONE_4PI_EPS0), sigma / 2, np.sqrt(4 * epsilon)] 

133 ) 

134 non_exclusion_exceptions = [] 

135 for index in range(nonbonded_force.getNumExceptions()): 

136 i, j, q1q2, sigma, epsilon = nonbonded_force.getExceptionParameters(index) 

137 q1q2, sigma, epsilon = map(_standardized, [q1q2, sigma, epsilon]) 

138 force.addExclusion(i, j) 

139 if q1q2 != 0.0 or epsilon != 0.0: 

140 non_exclusion_exceptions.append( 

141 (i, j, q1q2 * ONE_4PI_EPS0, sigma, 4 * epsilon) 

142 ) 

143 force.setForceGroup(force_group_index) 

144 system.addForce(force) 

145 if non_exclusion_exceptions: 

146 exceptions = openmm.CustomBondForce(f"step({rc}-r)*({potential})") 

147 for parameter in ["Qprod", "sigma", "eps4"]: 

148 exceptions.addPerBondParameter(parameter) 

149 for i, j, Qprod, sigma, eps4 in non_exclusion_exceptions: 

150 exceptions.addBond(i, j, [Qprod, sigma, eps4]) 

151 exceptions.setForceGroup(force_group_index) 

152 system.addForce(exceptions) 

153 nonbonded_force.setForceGroup(force_group_index + 1)

156class CustomIntegrator(openmm.CustomIntegrator): 

157 """An extension of OpenMM's CustomIntegrator_ class with an extra per-dof variable 

158 named `temperature`, whose content is the temperature of the heat bath associated to 

159 each degree of freedom. A per-dof temperature is necessary if the extended-space 

160 variables and the physical system are coupled adiabatically to thermostats at 

161 different temperatures. Otherwise, any other OpenMM integrator can be used. 

162 

163 Parameters 

164 ---------- 

165 temperature : float or unit.Quantity 

166 The temperature. 

167 step_size : float or unit.Quantity 

168 The step size with which to integrate the equations of motion. 

169 """ 

170 

171 def __init__(self, temperature, step_size): 

172 super().__init__(step_size) 

173 self.temperature = temperature 

174 self.addPerDofVariable("kT", unit.MOLAR_GAS_CONSTANT_R * temperature) 

175 self._up_to_date = False 

176 

177 def __repr__(self): 

178 """A human-readable version of each integrator step (adapted from openmmtools) 

179 

180 Returns 

181 ------- 

182 readable_lines : str 

183 A list of human-readable versions of each step of the integrator 

184 """ 

185 readable_lines = [] 

186 

187 self.getNumPerDofVariables() > 0 and readable_lines.append("Per-dof variables:") 

188 per_dof = [] 

189 for index in range(self.getNumPerDofVariables()): 

190 per_dof.append(self.getPerDofVariableName(index)) 

191 readable_lines.append(" " + ", ".join(per_dof)) 

192 

193 self.getNumGlobalVariables() > 0 and readable_lines.append("Global variables:") 

194 for index in range(self.getNumGlobalVariables()): 

195 name = self.getGlobalVariableName(index) 

196 value = self.getGlobalVariable(index) 

197 readable_lines.append(f" {name} = {value}") 

198 

199 readable_lines.append("Computation steps:") 

200 

201 step_type_str = [ 

202 "{target} <- {expr}", 

203 "{target} <- {expr}", 

204 "{target} <- sum({expr})", 

205 "constrain positions", 

206 "constrain velocities", 

207 "allow forces to update the context state", 

208 "if ({expr}):", 

209 "while ({expr}):", 

210 "end", 

211 ] 

212 indent_level = 0 

213 for step in range(self.getNumComputations()): 

214 line = "" 

215 step_type, target, expr = self.getComputationStep(step) 

216 if step_type == 8: 

217 indent_level -= 1 

218 command = step_type_str[step_type].format(target=target, expr=expr) 

219 line += "{:4d}: ".format(step) + " " * indent_level + command 

220 if step_type in [6, 7]: 

221 indent_level += 1 

222 readable_lines.append(line) 

223 return "\n".join(readable_lines)

225 def update_temperatures(self, system_temperature, extended_space_temperatures): 

226 nparticles = len(self.getPerDofVariableByName("kT")) - len( 

227 extended_space_temperatures 

228 ) 

229 temperatures = [system_temperature] * nparticles + extended_space_temperatures 

230 kT = [ 

231 unit.MOLAR_GAS_CONSTANT_R * T * openmm.Vec3(1, 1, 1) for T in temperatures 

232 ] 

233 self.setPerDofVariableByName("kT", kT) 

234 self._up_to_date = True 

235 

236 def step(self, steps): 

237 if not self._up_to_date: 

238 self.update_temperatures(self.temperature, []) 

239 super().step(steps)

242class AbstractMiddleRespaIntegrator(CustomIntegrator): 

243 """An abstract class for middle-type, multiple time-scale integrators. 

244 

245 .. warning:: 

246 This class is meant for inheritance only and does not actually include 

247 thermostatting. Concrete subclasses are available, such as 

248 :class:`MiddleMassiveNHCIntegrator` and :class:`GeodesicLangevinIntegrator`, 

249 for instance. 

250 

251 Child classes will differ by the thermostat algorithm, which must be implemented 

252 by overriding the `_bath` method (see the example below). 

253 Temperature is treated as a per-dof parameter so as to allow adiabatic simulations. 

254 

255 The following :term:`ODE` system is solved for every degree of freedom in the 

256 system, with possibly :math:`n_c` holonomic constraints and with forces possibly 

257 split into :math:`m` parts according to their characteristic time scales: 

258 

259 .. math:: 

260 & \\dot{r}_i = v_i \\\\ 

261 & \\dot{v}_i = \\frac{\\sum_{k=1}^m F_i^{[k]}}{m_i} 

262 + \\sum_{k=1}^{n_c} \\lambda_k \\nabla_{r_i} \\sigma_k 

263 + \\mathrm{bath}(T_i, v_i) \\\\ 

264 & \\sigma_k(\\mathbf{r}) = 0 

265 

266 An approximate solution is obtained by applying the Trotter-Suzuki splitting 

267 formula. In the particular case of two time scales, the default splitting scheme 

268 goes as follows: 

269 

270 .. math:: 

271 e^{\\Delta t\\mathcal{L}} = 

272 e^{\\frac{\\Delta t}{2}\\mathcal{L}^{[1]}_v} 

273 \\left[ 

274 e^{\\frac{\\Delta t}{2 n_0}\\mathcal{L}^{[0]}_v} 

275 \\left( 

276 e^{\\frac{\\Delta t}{2 n_0 n_b}\\mathcal{L}_r} 

277 e^{\\frac{\\Delta t}{n_0 n_b}\\mathcal{L}_\\mathrm{bath}} 

278 e^{\\frac{\\Delta t}{2 n_0 n_b}\\mathcal{L}_r} 

279 \\right)^{n_b} 

280 e^{\\frac{\\Delta t}{2 n_0}\\mathcal{L}^{[0]}_v} 

281 \\right]^{n_0} 

282 e^{\\frac{\\Delta t}{2}\\mathcal{L}^{[1]}_v} 

283 

284 Each exponential operator is the solution of a particular subsystem of equations. 

285 If :math:`\\mathrm{bath}(T_i, v_i) = 0`, the scheme above is time-reversible, 

286 measure-preserving, and symplectic. It is referred to as the ``VV-Middle`` scheme 

287 :cite:`Zhang_2019`, where VV stands for Velocity Verlet. An alternative approach 

288 is also available, which is: 

289 

290 .. math:: 

291 e^{\\Delta t\\mathcal{L}} = 

292 \\left[ 

293 \\left( 

294 e^{\\frac{\\Delta t}{2 n_0 n_b}\\mathcal{L}_r} 

295 e^{\\frac{\\Delta t}{n_0 n_b}\\mathcal{L}_\\mathrm{bath}} 

296 e^{\\frac{\\Delta t}{2 n_0 n_b}\\mathcal{L}_r} 

297 \\right)^{n_b} 

298 e^{\\frac{\\Delta t}{n_0}\\mathcal{L}^{[0]}_v} 

299 \\right]^{n_0} 

300 e^{\\Delta t \\mathcal{L}^{[1]}_v} 

301 

302 This is referred to as the ``LF-Middle`` scheme :cite:`Zhang_2019`, where LF stands 

303 for Leap-Frog. In contrast to the previous scheme, it is not time-reversible. 

304 However, in single time-scale simulations, the two approaches result in equivalent 

305 coordinate trajectories, while the latter provides a velocity trajectory more 

306 consistent with the Maxwell-Boltzmann distribution at the specified temperature 

307 :cite:`Zhang_2019`. 

308 

309 Parameters 

310 ---------- 

311 temperature : unit.Quantity 

312 The temperature of the heat bath. 

313 step_size : float or unit.Quantity 

314 The outer step size with which to integrate the equations of motion. 

315 

316 Keyword Args 

317 ------------ 

318 num_rattles : int, default=0 

319 The number of RATTLE computations for geodesic integration 

320 :cite:`Leimkuhler_2016`. If ``num_rattles=0``, then no constraints are 

321 considered at all. 

322 scheme : str, default='VV-Middle' 

323 Which splitting scheme will be used. Valid options are 'VV-Middle' and 

324 'LF-Middle'. 

325 respa_loops : list(int), default=[1] 

326 A list of `m` integers, where `respa_loops[k]` determines how many substeps 

327 with force group `k` are internally executed for every step with force group 

328 `k+1`. 

329 bath_loops : int, default=1 

330 The number of iterations of the bath operator per each step at time scale 

331 `0`. This is useful when the bath operator is not exact, but derived from a 

332 splitting solution. 

333 embodied_force_groups : list(int), default=[] 

334 A list of indices of force groups. The presence of an index `i` is this list 

335 means that the contribution of force group `i` is embodied in force group 

336 `i+1`. Therefore, such contribution must be properly subtracted during the 

337 integration at time scale `i+1`. This feature requires OpenMM 7.5 or a newer 

338 version. 

339 unroll_loops : bool, default=True 

340 Whether the integrator loops should be unrolled for improving efficiency. 

341 Using ``unroll_loops=False`` can be useful for printing the integrator 

342 steps. 

343 

344 Example 

345 ------- 

346 >>> from ufedmm import integrators 

347 >>> from openmm import unit 

348 >>> class MiddleNoseHooverIntegrator(integrators.AbstractMiddleRespaIntegrator): 

349 ... def __init__(self, ndof, tau, temperature, step_size, num_rattles=1): 

350 ... super().__init__( 

351 ... temperature, step_size, num_rattles, 'VV-Middle', [1], 1 

352 ... ) 

353 ... kB = 8.3144626E-3*unit.kilojoules_per_mole/unit.kelvin 

354 ... gkT = ndof*unit.MOLAR_GAS_CONSTANT_R*temperature 

355 ... self.addGlobalVariable('gkT', gkT) 

356 ... self.addGlobalVariable('Q', gkT*tau**2) 

357 ... self.addGlobalVariable('v_eta', 0) 

358 ... self.addGlobalVariable('twoK', 0) 

359 ... self.addGlobalVariable('scaling', 1) 

360 ... def _bath(self, fraction): 

361 ... self.addComputeSum('twoK', 'm*v*v') 

362 ... self.addComputeGlobal( 

363 ... 'v_eta', f'v_eta + {0.5*fraction}*dt*(twoK - gkT)/Q' 

364 ... ) 

365 ... self.addComputeGlobal('scaling', f'exp(-{fraction}*dt*v_eta)') 

366 ... self.addComputePerDof('v', f'v*scaling') 

367 ... self.addComputeGlobal( 

368 ... 'v_eta', f'v_eta + {0.5*fraction}*dt*(scaling^2*twoK - gkT)/Q' 

369 ... ) 

370 >>> integrator = MiddleNoseHooverIntegrator( 

371 ... 500, 10*unit.femtoseconds, 300*unit.kelvin, 

372 ... 1*unit.femtoseconds, num_rattles=0 

373 ... ) 

374 >>> print(integrator) 

375 Per-dof variables: 

376 kT 

377 Global variables: 

378 gkT = 1247.169392722986 

379 Q = 0.1247169392722986 

380 v_eta = 0.0 

381 twoK = 0.0 

382 scaling = 1.0 

383 Computation steps: 

384 0: allow forces to update the context state 

385 1: v <- v + 0.5*dt*f/m 

386 2: x <- x + 0.5*dt*v 

387 3: twoK <- sum(m*v*v) 

388 4: v_eta <- v_eta + 0.5*dt*(twoK - gkT)/Q 

389 5: scaling <- exp(-1.0*dt*v_eta) 

390 6: v <- v*scaling 

391 7: v_eta <- v_eta + 0.5*dt*(scaling^2*twoK - gkT)/Q 

392 8: x <- x + 0.5*dt*v 

393 9: v <- v + 0.5*dt*f/m 

394 """ 

395 

396 def __init__( 

397 self, 

398 temperature, 

399 step_size, 

400 num_rattles=0, 

401 scheme="VV-Middle", 

402 respa_loops=[1], 

403 bath_loops=1, 

404 intertwine=True, 

405 embodied_force_groups=[], 

406 unroll_loops=True, 

407 ): 

408 if scheme not in ["LF-Middle", "VV-Middle"]: 

409 raise Exception(f"Invalid value {scheme} for keyword scheme") 

410 super().__init__(temperature, step_size) 

411 self._num_rattles = num_rattles 

412 self._scheme = scheme 

413 self._respa_loops = respa_loops 

414 self._bath_loops = bath_loops 

415 self._intertwine = intertwine 

416 self._subtractive_groups = embodied_force_groups 

417 num_rattles > 0 and self.addPerDofVariable("x0", 0) 

418 num_rattles > 1 and self.addGlobalVariable("irattle", 0) 

419 if not unroll_loops: 

420 for scale, n in enumerate(respa_loops): 

421 n > 1 and self.addGlobalVariable(f"irespa{scale}", 0) 

422 bath_loops > 1 and self.addGlobalVariable("ibath", 0) 

423 if embodied_force_groups: 

424 if openmm.__version__ < "7.5": 

425 raise Exception( 

426 "Use of `embodied_force_groups` option requires OpenMM >= 7.5" 

427 ) 

428 self.addPerDofVariable("f_emb", 0) 

429 integration_groups = set(range(len(respa_loops))) - set( 

430 embodied_force_groups 

431 ) 

432 self.setIntegrationForceGroups(integration_groups) 

433 

434 self.addUpdateContextState() 

435 self._step_initialization() 

436 if unroll_loops: 

437 self._integrate_respa_unrolled(1, len(respa_loops) - 1) 

438 else: 

439 self._integrate_respa(1, len(respa_loops) - 1)

441 def _step_initialization(self): 

442 pass 

443 

444 def _integrate_respa(self, fraction, scale): 

445 if scale >= 0: 

446 n = self._respa_loops[scale] 

447 if n > 1: 

448 self.addComputeGlobal(f"irespa{scale}", "0") 

449 self.beginWhileBlock(f"irespa{scale} < {n-1/2}") 

450 self._boost(fraction / (2 * n if self._scheme == "VV-Middle" else n), scale) 

451 self._integrate_respa(fraction / n, scale - 1) 

452 self._scheme == "VV-Middle" and self._boost(fraction / (2 * n), scale) 

453 if n > 1: 

454 self.addComputeGlobal(f"irespa{scale}", f"irespa{scale} + 1") 

455 self.endBlock() 

456 else: 

457 self._intertwine or self._translation(0.5 * fraction) 

458 n = self._bath_loops 

459 if n > 1: 

460 self.addComputeGlobal("ibath", "0") 

461 self.beginWhileBlock(f"ibath < {n-1/2}") 

462 self._intertwine and self._translation(0.5 * fraction / n) 

463 self._bath(fraction / n) 

464 self._num_rattles > 0 and self.addConstrainVelocities() 

465 self._intertwine and self._translation(0.5 * fraction / n) 

466 if n > 1: 

467 self.addComputeGlobal("ibath", "ibath + 1") 

468 self.endBlock() 

469 self._intertwine or self._translation(0.5 * fraction) 

470 

471 def _integrate_respa_unrolled(self, fraction, scale): 

472 if scale >= 0: 

473 n = self._respa_loops[scale] 

474 for i in range(n): 

475 self._boost( 

476 fraction / (2 * n if self._scheme == "VV-Middle" and i == 0 else n), 

477 scale, 

478 ) 

479 self._integrate_respa_unrolled(fraction / n, scale - 1) 

480 self._scheme == "VV-Middle" and i == n - 1 and self._boost( 

481 fraction / (2 * n), scale 

482 ) 

483 else: 

484 n = self._bath_loops 

485 self._intertwine or self._translation(0.5 * fraction) 

486 for i in range(n): 

487 self._intertwine and self._translation( 

488 fraction / (2 * n if i == 0 else n) 

489 ) 

490 self._bath(fraction / n) 

491 self._num_rattles > 0 and self.addConstrainVelocities() 

492 i == n - 1 and self._intertwine and self._translation( 

493 fraction / (2 * n) 

494 ) 

495 self._intertwine or self._translation(0.5 * fraction) 

496 

497 def _translation(self, fraction): 

498 if self._num_rattles > 1: 

499 self.addComputeGlobal("irattle", "0") 

500 self.beginWhileBlock(f"irattle < {self._num_rattles-1/2}") 

501 self.addComputePerDof("x", f"x + {fraction/max(1, self._num_rattles)}*dt*v") 

502 if self._num_rattles > 0: 

503 self.addComputePerDof("x0", "x") 

504 self.addConstrainPositions() 

505 self.addComputePerDof( 

506 "v", f"v + (x - x0)/({fraction/self._num_rattles}*dt)" 

507 ) 

508 self.addConstrainVelocities() 

509 if self._num_rattles > 1: 

510 self.addComputeGlobal("irattle", "irattle + 1") 

511 self.endBlock() 

512 

513 def _boost(self, fraction, scale): 

514 if len(self._respa_loops) > 1: 

515 if scale - 1 in self._subtractive_groups: 

516 self.addComputePerDof("f_emb", f"f{scale-1}") 

517 self.addComputePerDof("v", f"v + {fraction}*dt*(f{scale}-f_emb)/m") 

518 else: 

519 self.addComputePerDof("v", f"v + {fraction}*dt*f{scale}/m") 

520 else: 

521 self.addComputePerDof("v", f"v + {fraction}*dt*f/m") 

522 self._num_rattles > 0 and self.addConstrainVelocities() 

523 

524 def _bath(self, fraction): 

525 return

528class GeodesicLangevinIntegrator(AbstractMiddleRespaIntegrator): 

529 """A geodesic Langevin integrator :cite:`Leimkuhler_2016`, which can be integrated 

530 by using either the LF-Middle or the VV-Middle scheme :cite:`Zhang_2019`. 

531 

532 .. note: 

533 The VV-Middle scheme is also known as the BAOAB :cite:`Leimkuhler_2016` method. 

534 

535 Parameters 

536 ---------- 

537 temperature : float or unit.Quantity 

538 The temperature. 

539 friction_coefficient : float or unit.Quantity 

540 The friction coefficient. 

541 step_size : float or unit.Quantity

801class RegulatedNHLIntegrator(AbstractMiddleRespaIntegrator): 

802 """A regulated version of the massive Nose-Hoover-Langevin 

803 :cite:`Samoletov_2007,Leimkuhler_2009` method. Regulation means that the system 

804 Hamiltonian is modified so that velocities remain below a temperature-dependent 

805 limit. This is closely related to the SIN(R) method :cite:`Leimkuhler_2013` and 

806 allows multiple time-scale integration with very large outer time steps, without 

807 resonance. 

808 

809 .. info: 

810 If `regulation_parameter = 1` (default), this method is equivalent to SIN(R) 

811 with a single thermostat per degree of freedom (that is, `L=1`). 

812 

813 The following :term:`SDE` system is solved for every degree of freedom in the 

814 system: 

815 

816 .. math:: 

817 & dr_i = v_i dt \\\\ 

818 & dp_i = F_i dt - v_{\\eta_i} m_i v_i dt \\\\ 

819 & dv_{\\eta_i} = \\frac{1}{Q}\\left(\\frac{n+1}{n} m_i v_i^2 - k_B T\\right) dt 

820 - \\gamma v_{\\eta_i} dt + \\sqrt{\\frac{2\\gamma k_B T}{Q}} dW_i, 

821 

822 where: 

823 

824 .. math:: 

825 v_i = c_i \\tanh\\left(\\frac{p_i}{m_i c_i}\\right). 

826 

827 Here, :math:`n` is the regulation parameter and :math:`c_i` is the maximum speed for 

828 degree of freedom `i`, given by :math:`c_i = \\sqrt{\\frac{n k T}{m_i}}`. The 

829 inertial parameter :math:`Q` is defined as :math:`Q = n k_B T \\tau^2`, with 

830 :math:`\\tau` being a relaxation time :cite:`Tuckerman_1992`. An approximate 

831 solution is obtained by applying the Trotter-Suzuki splitting formula: 

832 

833 .. math:: 

834 e^{\\Delta t\\mathcal{L}} = 

835 e^{\\frac{\\Delta t}{2}\\mathcal{L}^1_p} 

836 \\left[e^{\\frac{\\delta t}{2}\\mathcal{L}^0_p} 

837 e^{\\frac{\\delta t}{2}\\mathcal{L}_r} 

838 e^{\\delta t \\mathcal{L}_\\mathrm{bath}} 

839 e^{\\frac{\\delta t}{2}\\mathcal{L}_r} 

840 e^{\\frac{\\delta t}{2}\\mathcal{L}^0_p}\\right]^m 

841 e^{\\frac{\\Delta t}{2}\\mathcal{L}^1_p} 

842 

843 where :math:`\\delta t = \\frac{\\Delta t}{m}`. Each exponential operator above is 

844 the solution of a differential equation. 

845 

846 The exact solution for the physical-system part is: 

847 

848 .. math:: 

849 r_i(t) = r_i^0 + c_i \\mathrm{tanh}\\left(\\frac{p_i}{m c_i}\\right) t 

850 

851 .. math:: 

852 p_i(t) = p_i^0 + F_i t 

853 

854 The bath propagator is further split as: 

855 

856 .. math:: 

857 e^{\\delta t \\mathcal{L}_\\mathrm{bath}} = 

858 e^{\\frac{\\delta t}{2m}\\mathcal{L}_B} 

859 e^{\\frac{\\delta t}{2m}\\mathcal{L}_S} 

860 e^{\\frac{\\delta t}{m}\\mathcal{L}_O} 

861 e^{\\frac{\\delta t}{2m}\\mathcal{L}_S} 

862 e^{\\frac{\\delta t}{2m}\\mathcal{L}_B} 

863 

864 Part 'B' is a boost, whose solution is: 

865 

866 .. math:: 

867 v_{\\eta_i}(t) = v_{\\eta_i}^0 + 

868 \\frac{1}{Q}\\left(\\frac{n+1}{n} m_i v_i^2 - k_B T\\right) t 

869 

870 Part 'S' is a scaling, whose solution is: 

871 

872 .. math:: 

873 p_i(t) = m_i c_i \\mathrm{arcsinh}\\left[ 

874 \\sinh\\left(\\frac{p_i^0}{m_i c_i}\\right) e^{- v_{\\eta_i} t} 

875 \\right] 

876 

877 Part 'O' is an Ornstein–Uhlenbeck process, whose solution is: 

878 

879 .. math:: 

880 v_{\\eta_i}(t) = v_{\\eta_i}^0 e^{-\\gamma t} 

881 + \\sqrt{\\frac{k_B T}{Q}(1-e^{-2\\gamma t})} R_N 

882 

883 where :math:`R_N` is a normally distributed random number. 

884 

885 Parameters 

886 ---------- 

887 step_size : float or unit.Quantity 

888 The outer step size with which to integrate the equations of motion. 

889 loops : int 

890 The number of internal substeps at each time step. 

891 temperature : unit.Quantity 

892 The temperature of the heat bath. 

893 time_scale : unit.Quantity (time) 

894 The relaxation time (:math:`\\tau`) of the Nose-Hoover thermostat. 

895 friction_coefficient : unit.Quantity (1/time) 

896 The friction coefficient (:math:`\\gamma`) of the Langevin thermostat. 

897 regulation_parameter : int or float 

898 The regulation parameter n. 

899 

900 Keyword Args 

901 ------------ 

902 semi_regulated : bool, default=True 

903 Whether to use the semi-regulated NHL version of the method instead of its 

904 fully-regulated version. 

905 split_ornstein_uhlenbeck : bool, default=True 

906 Whether to split the drifted Ornstein-Uhlenbeck operator. 

907 **kwargs 

908 All keyword arguments in :class:`AbstractMiddleRespaIntegrator`, except 

909 ``num_rattles``. 

910 """ 

911
